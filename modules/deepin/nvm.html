安装 nvm ​

1. 下载 NVM 安装脚本 ​

bash
wget https://github.com/nvm-sh/nvm/archive/refs/tags/v0.39.5.tar.gz

2. 解压文件并进入目录 ​

bash
tar -xzf v0.39.5.tar.gz
cd nvm-0.39.5

3. 运行安装脚本 ​

bash
./install.sh

4. 更新终端配置 ​

bash
source ~/.bashrc

5. 验证安装 ​

bash
nvm --version

6. 安装 nodejs ​

bash
nvm install <version>

7. 设置镜像 ​

bash
# 编辑 ~/.bashrc
vim ~/.bashrc

# 文件最后加上:
export NVM_NODEJS_ORG_MIRROR=https://npmmirror.com/mirrors/node/

#保存
:wq

# 添加完成后，重新加载配置文件
source ~/.bashrc

# 验证
nvm ls-remote
